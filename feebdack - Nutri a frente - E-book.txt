function Particle() {
  this.reset = function() {
    this.x = Math.random() * W;
    this.y = Math.random() * H;
    this.r = Math.random() * 1.5 + .3;
    this.vx = (Math.random() - .5) * .3;
    this.vy = (Math.random() - .5) * .3;
    this.life = 0;
    this.maxLife = Math.random() * 300 + 150;
    this.hue = Math.random() * 30 + 200; // blue range
  };
  this.reset();
}

  const p = new Particle();
  p.life = Math.random() * p.maxLife; // stagger start
  particles.push(p);

// Glow orbs
const orbs = [
  { x: .2, y: .3, r: 320, color: 'rgba(13,59,110,0.22)' },
  { x: .8, y: .7, r: 280, color: 'rgba(26,127,232,0.12)' },
  { x: .5, y: .1, r: 200, color: 'rgba(63,169,255,0.07)' },
];
 
function drawOrbs() {
  orbs.forEach(o => {
    const t = Date.now() * .0003;
    const ox = Math.sin(t + o.x * 10) * 30;
    const oy = Math.cos(t * .7 + o.y * 10) * 20;
    const grd = ctx.createRadialGradient(
      o.x * W + ox, o.y * H + oy, 0,
      o.x * W + ox, o.y * H + oy, o.r
    );
    grd.addColorStop(0, o.color);
    grd.addColorStop(1, 'transparent');
    ctx.fillStyle = grd;
    ctx.fillRect(0, 0, W, H);
  });
}
 
function animate() {
  ctx.clearRect(0, 0, W, H);
  drawOrbs();
 
  particles.forEach(p => {
    p.life++;
    if (p.life > p.maxLife) p.reset();
    const alpha = Math.sin(Math.PI * p.life / p.maxLife) * .55;
    ctx.beginPath();
    ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
    ctx.fillStyle = `hsla(${p.hue}, 90%, 75%, ${alpha})`;
    ctx.fill();
    p.x += p.vx; p.y += p.vy;
    if (p.x < 0 || p.x > W) p.vx *= -1;
    if (p.y < 0 || p.y > H) p.vy *= -1;

  requestAnimationFrame(animate);
}
